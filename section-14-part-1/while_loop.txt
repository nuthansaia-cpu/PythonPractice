# Sum of Digits of a Number.

# Reverse a Number

n=1257
rev=0
while n>0 :
      r=n%10
      rev=rev * 10 + r 
      n=n//10
print('Reverse Number:',rev)

print("----------------------------------------------------------")

# Check a No is a Palindrome

n=1221
rev=0
m=n
while n>0 :
      r=n%10
      rev=rev * 10 + r 
      n=n//10
if rev==m:
     print(m,'is a palindrome number')
else:
     print(m,'is not a palindrome number ')
print("----------------------------------------------------------")
